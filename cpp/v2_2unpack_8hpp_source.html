132 inline bool unpacker::flush_zone()
133 {
134  if(referenced()) {
135  try {
136  m_z->push_finalizer(&detail::decr_count, get_raw_buffer());
137  } catch (...) {
138  return false;
139  }
140  set_referenced(false);
141 
142  detail::incr_count(get_raw_buffer());
143  }
144 
145  return true;
